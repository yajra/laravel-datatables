 418:      *

 421:     public function tableNames()
 422:     {
 423:         $names = [];
 424:         $query = $this->getBuilder();
 425:         $names[] = $query->from;
 426:         $joins = $query->joins ?: [];
 427:         $databasePrefix = $this->databasePrefix();
 428:         foreach ($joins as $join) {
 429:             $table = preg_split("/ as /i", $join->table);
 430:             $names[] = $table[0];
 431:             if (isset($table[1]) && ! empty($databasePrefix) && strpos($table[1], $databasePrefix) == 0) {
 432:                 $names[] = preg_replace('/^' . $databasePrefix . '/', '', $table[1]);

 475:      * @return null

 477:     public function doOrdering()
 478:     {
 479:         if (array_key_exists('order', $this->input) && count($this->input['order']) > 0) {
 480:             for ($i = 0, $c = count($this->input['order']); $i < $c; $i++) {
 481:                 $order_col = (int) $this->input['order'][$i]['column'];
 482:                 $order_dir = $this->input['order'][$i]['dir'];
 483:                 $column = $this->input['columns'][$order_col];
 484: 
 485:                 if ($column['orderable'] <> "true") {
 486:                     continue;
 487:                 }
 488: 
 489:                 if (isset($column['name']) and $column['name'] <> '') {
 490:                     $this->query->orderBy($column['name'], $order_dir);
 491:                 } else {
 492:                     $this->query->orderBy($this->columns[$order_col], $order_dir);
 493:                 }
 494:             }

 499:      * Places extra columns

 501:     public function initColumns()
 502:     {
 503:         foreach ($this->result_array as $rkey => &$rvalue) {
 504: 
 505:             // Convert data array to object value
 506:             $data = [];
 507:             foreach ($rvalue as $key => $value) {
 508:                 if (is_object($this->result_object[$rkey])) {
 509:                     $data[$key] = $this->result_object[$rkey]->$key;
 510:                 } else {
 511:                     $data[$key] = $value;
 512:                 }
 513:             }
 514: 
 515:             // Process add columns
 516:             foreach ($this->extra_columns as $key => $value) {
 517:                 if (is_string($value['content'])):
 518:                     $value['content'] = $this->compileBlade($value['content'], $data);
 519:                 elseif (is_callable($value['content'])):
 520:                     $value['content'] = $value['content']($this->result_object[$rkey]);
 521:                 endif;
 522: 
 523:                 $rvalue = $this->includeInArray($value, $rvalue);
 524:             }
 525: 
 526:             // Process edit columns
 527:             foreach ($this->edit_columns as $key => $value) {
 528:                 if (is_string($value['content'])):
 529:                     $value['content'] = $this->compileBlade($value['content'], $data);
 530:                 elseif (is_callable($value['content'])):
 531:                     $value['content'] = $value['content']($this->result_object[$rkey]);
 532:                 endif;
 533: 
 534:                 $rvalue[$value['name']] = $value['content'];
 535:             }

 542:      * @param $str
 543:      * @param array $data
 544:      * @return string
 545:      * @throws \Exception
 546:      */
 547:     public function compileBlade($str, $data = [])
 548:     {
 549:         $empty_filesystem_instance = new Filesystem;
 550:         $blade = new BladeCompiler($empty_filesystem_instance, 'datatables');
 551:         $parsed_string = $blade->compileString($str);
 552: 
 553:         ob_start() and extract($data, EXTR_SKIP);
 554: 
 555:         try {
 556:             eval('?>' . $parsed_string);
 557:         } catch (\Exception $e) {
 558:             ob_end_clean();
 559:             throw $e;
 560:         }
 561: 
 562:         $str = ob_get_contents();

 569:      * Places item of extra columns into result_array by care of their order

 571:      * @param $item

 574:      */
 575:     public function includeInArray($item, $array)
 576:     {
 577:         if ($item['order'] === false) {
 578:             return array_merge($array, [$item['name'] => $item['content']]);
 579:         } else {
 580:             $count = 0;
 581:             $last = $array;
 582:             $first = [];
 583:             foreach ($array as $key => $value) {
 584:                 if ($count == $item['order']) {
 585:                     return array_merge($first, [$item['name'] => $item['content']], $last);
 586:                 }
 587: 
 588:                 unset($last[$key]);
 589:                 $first[$key] = $value;
 590: 
 591:                 $count++;
 592:             }
 593:         }
